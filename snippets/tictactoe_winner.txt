function winner(board) {
	const WINNING_LINES = [
		[[0, 0], [0, 1], [0, 2]],
		[[1, 0], [1, 1], [1, 2]],
		[[2, 0], [2, 1], [2, 2]],
		[[0, 0], [1, 0], [2, 0]],
		[[0, 1], [1, 1], [2, 1]],
		[[0, 2], [1, 2], [2, 2]],
		[[0, 0], [1, 1], [2, 2]],
		[[0, 2], [1, 1], [2, 0]],
	];
	for (const line of WINNING_LINES) {
		const values = line.map(([r, c]) => board[r][c]);
		let first = values[0]
		if (values.every(s => s !== 0 && s === first)) {
			return first;
		}
	}
	return 0;
}
